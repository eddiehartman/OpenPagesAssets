select log.DB2_FUNC as Function, log.EVIDENCE_ID, log.EVIDENCE_NAME , log.EVIDENCE_TYPE ,
  CASE WHEN evidence.delayed =  'N' THEN 'No'
  ELSE 'Yes'
  END AS evidence_delayed,
  CASE WHEN evidence.exception =  'N' THEN 'No'
  ELSE 'Yes'
  END AS evidence_exception,
  CASE WHEN evidence.extra_type =  1 THEN 'PRODUCTION'
  ELSE 'OBSOLETE'
  END AS Evidence_Status, evidence.comment,
  log.DOCUMENT_ID, log.DOCUMENT_NAME, log.DOCUMENT_REVISION,
  CASE WHEN log.ATTACHMENT_FORMAT =  1 THEN 'MAIL'
  ELSE 'FILE'
  END AS format,
  log.ACTION as Operation, log.TS, log.OLD_VALUE, log.NEW_VALUE, log.NOTES_EMAIL from epolicy.L_EVIDENCE log
  inner join epolicy.evidence as evidence on (evidence.id = log.evidence_id)
  where log.document_id = '{id}' and log.document_revision = '{revision}'  and log.document_revision_count = '{revision_count}'
