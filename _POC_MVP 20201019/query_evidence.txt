select distinct evidence.id, evidence.name, evidence_type.id as type_id,
  evidence_type.name as type_name, xdocument.id as document_id,
  xdocument.name as document_name, evidence.format, evidence.delayed,
  evidence.exception, evidence.extra_type    , evidence.comment, evidence.updated_by,   evidence.updated_ts,
  CASE WHEN evidence.extra_type =  1 THEN 'Production'
  ELSE 'Obsolete'
  END AS Evidence_Status
  from epolicy.evidence as evidence
  inner join epolicy.evidence_type as evidence_type on evidence_type.id = evidence.evidence_type_id
  inner join epolicy.evidence_xdocument as evidence_xdocument on evidence_xdocument.evidence_id = evidence.id
  inner join epolicy.all_xdocuments as xdocument on xdocument.id = evidence_xdocument.xdocument_id
  and xdocument.revision = evidence_xdocument.xdocument_revision and xdocument.revision_count = evidence_xdocument.xdocument_revision_count
  and xdocument.id =  '{id}'
