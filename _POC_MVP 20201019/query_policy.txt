    select distinct xdocument.id, xdocument.name, xdocument.file_name,
    account.shortname, account.id as account_id, user.EMAIL,
    xdocument.template_id, template_type.name as template_type_name,
    xdocument.template_type_id, policy_type.name as policy_type_name,
    policy.policy_type_id, xdocument.status_id, status_type.status,policy.current_approval_date,policy.NEXT_REVIEW_DATE,
    xdocument.locked_flag, xdocument.locked_by, policy.account_number,
    account_number_type.name account_number_type_name, template.format, xdocument.template_version, policy.id as policy_id,
    xdocument.environment, xdocument.locked_ts, account.location_id, location.full_name, xdocument.history_flag, xdocument.updated_ts, xdocument.deleted_ts,
    xdocument.revision, template_policyType.name as template_policyType_name, template_policyType.id as template_policyType_id,
    policy.id as policy_id, xdocument.imported_flag as imported_flag , xdocument.revision_count as revision_count
    from epolicy.all_xdocuments xdocument inner join epolicy.policy policy on (policy.id = xdocument.policy_id)
    inner join epolicy.policy_type as policy_type on (policy.policy_type_id = policy_type.id)
    inner join epolicy.account as account on (policy.account_id = account.id)
    inner join epolicy.template_type as template_type on (xdocument.template_type_id = template_type.id)
    left outer join epolicy.account_number_type as account_number_type on policy.account_number_type_id = account_number_type.id
    left outer join epolicy.status_type as status_type on (xdocument.status_id = status_type.id)
    left outer join epolicy.account_user as account_user on (account_user.account_id = account.id)
    left outer join epolicy.xdocument_user as xdocument_user on (xdocument_user.xdocument_id = xdocument.id)
    left outer join epolicy.user as user on (user.id = xdocument.owner_id)
    left outer join epolicy.location_admin_tree as rpl on (rpl.child_id = account.location_id)
    left outer join epolicy.template as template on (template.id = xdocument.template_id)
    left outer join epolicy.policy_type as template_policyType on (template_policyType.id = template.policy_type_id)
    inner join epolicy.location as location on (account.location_id = location.id)
    left outer join epolicy.policy_threat as policy_threat on (policy_threat.policy_id = policy.id)
    where xdocument.is_deleted ='N' and template_type.name = 'Base Policy' and policy_type.name = 'CSD'  and account.name = '{name}'
